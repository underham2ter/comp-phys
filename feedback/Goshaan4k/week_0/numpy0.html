def checkerboard(w, b):
    m = np.zeros((8,8),dtype=str)
    m[1::2,::2] = b
    m[::2,1::2] = b
    m[1::2,1::2] = w
    m[::2,::2] = w
    return m

def cartesian_to_polar(c):
    p = np.zeros((10, 2), dtype=float)
    p[:, 0] = np.sqrt((c[:, 0])**2 + (c[:, 1])**2)
    p[:, 1] = np.arctan((c[:, 1]/c[:, 0]))
    return p

def make_curve(n, a):
    phi = np.linspace(1, 10, num=n)
    r = a + np.cos(phi)
    xy = np.empty((n, 2))
    xy[:, 0] = r * np.cos(phi)
    xy[:, 1] = r * np.sin(phi)
    return xy
